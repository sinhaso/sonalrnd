#set up nfs server on Ubuntu (NFS server 192.168.0.108)

sudo mkdir /nfs_shared_vol

cd sudo chmod 777 /nfs_shared_vol

cd /nfs_shared_vol

cd sudo chmod 777 *

sudo apt-get update
sudo apt-get install nfs-kernel-server
sudo nano /etc/exports
        /nfs_shared_vol *(rw,sync,subtree_check,insecure)

sudo vi /etc/hosts.allow
#empty

sudo vi /etc/hosts.deny
#empty

chown nobody:nogroup /nfs_shared_vol

exportfs -a
sudo /etc/init.d/nfs-kernel-server restart
sudo /etc/init.d/portmap restart
